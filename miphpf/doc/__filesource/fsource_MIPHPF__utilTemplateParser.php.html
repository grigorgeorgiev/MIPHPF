  239.         /**
  240.          * Parses a subsection
  241.          * 
  242.          * @param array $templateSections 
  243.          * @param array $sectionInfos 
  244.          * @param int $subsectionIndex (optional)
  245.          * @return string the parsed string
  246.          */
  247.         protected function templateParseSubsection($templateSections, $sectionInfos, $subsectionIndex = 0)
  248.         {
  249.             $html = '';
  250.             foreach ($templateSections as $templateSection) {
  251.                 $name = $templateSection['name'];
  252.                 
  253.                 // If it is an end just add the contents
  254.                 if ($name == 'END') {
  255.                     $html .= $templateSection['contents'];
  256.                     continue;
  257.                 }
  258.                 
  259.                 $html .= $templateSection['prefix'];
  260.                 
  261.                 // Find the section info
  262.                 if (empty($sectionInfos[$name]))
  263.                     continue;
  264.                 $sectionInfo = $sectionInfos[$name];
  265.                 if (is_array($sectionInfo)) {
  266.                     $sectionInfo = $sectionInfo[$subsectionIndex];
  267.                 }
  268.                 
  269.                 // Check if we have subsections that are arrays
  270.                 $hasSubsectionsArray = false;
  271.                 foreach ($sectionInfo->_subsections as $subsection) {
  272.                     if (is_array($subsection)) {
  273.                         $hasSubsectionsArray = true;
  274.                         break;
  275.                     }
  276.                 }
  277.                 
  278.                 if (!$hasSubsectionsArray) {
  279.                     if (isset($templateSection['subsections']))
  280.                         $contents = $this->templateParseSubsection($templateSection['subsections'], $sectionInfo->_subsections);
  281.                     else
  282.                         $contents = $templateSection['contents'];
  283.                 } else
  284.                     $contents = '';
  285.                 
  286.                 if ($sectionInfo->_iterationsCount > self::TEMPLATE_PARSER_PREPROCESS_AT_ITERATIONS) {
  287.                     // If more than 5 iterations do preprocessing -
  288.                     // split the array variables from normal variables
  289.                     $arrayValues = $arrayKeys = $normalValues = $normalKeys = array();
  290.                     foreach ($sectionInfo->_values as $key => $r) {
  291.                         if (is_array($r)) {
  292.                             $arrayValues[] = $r;
  293.                             $arrayKeys[] = $key;
  294.                         } else {
  295.                             $normalValues[] = $r;
  296.                             $normalKeys[] = $key;
  297.                         }
  298.                     }
  299.                     $search = array_merge($normalKeys, $arrayKeys);
  300.                     
  301.                     for ($i = 0; $i < $sectionInfo->_iterationsCount; $i++) {
  302.                         
  303.                         $replace = $normalValues;
  304.                         foreach ($arrayValues as $r)
  305.                             $replace[] = $r[$i];
  306.                         
  307.                         if ($hasSubsectionsArray)
  308.                             $contents = $this->templateParseSubsection($templateSection['subsections'], $sectionInfo->_subsections, $i);
  309.                         $html .= str_replace($search, $replace, $contents);
  310.                     }
  311.                 } else {
  312.                     $search = array_keys($sectionInfo->_values);
  313.                     for ($i = 0; $i < $sectionInfo->_iterationsCount; $i++) {
  314.                         
  315.                         $replace = array();
  316.                         foreach ($sectionInfo->_values as $r)
  317.                             $replace[] = is_array($r) ? $r[$i] : $r;
  318.                         
  319.                         if ($hasSubsectionsArray and isset($templateSection['subsections']))
  320.                             $contents = $this->templateParseSubsection($templateSection['subsections'], $sectionInfo->_subsections, $i);
  321.                         $html .= str_replace($search, $replace, $contents);
  322.                     }
  323.                 }
  324.             }
  325.             
  326.             return $html;
  327.         }

  341.         /**
  342.          * Parse a template section recursive method
  343.          * 
  344.          * @param int $lastTagEnd 
  345.          * @param int $startPos 
  346.          * @param array $sections 
  347.          * @access protected
  348.          */
  349.         protected function parseSingleSection($lastTagEnd, $startPos, &$sections)
  350.         {
  351.             $theTemplateContents = &$this->_templateFileContents;
  352.             
  353. //            $nameStart = 0; // name start
  354. //            $nameEnd = 0; // This is start of end of the start tag
  355. //            $nextTagStart = 0; // The points to the current tag beginning - to the mi:section
  356. //            $pos = 0;    // The end of the last tag
  357.             
  358.             $pos = $startPos;
  359.             $inSection = false;
  360.             $section = array();
  361.  
  362.             do {
  363.                 // Next tag start
  364.                 $nextTagStart = strpos($theTemplateContents, self::TEMPLATE_PARSER_SECTION_TAG, $pos);
  365.                 if ($nextTagStart === false) {
  366.                     $section['name'] = 'END';
  367.                     $section['contents'] = substr($theTemplateContents, $pos);
  368.                     $sections[] = $section;
  369.                     break;
  370.                 }
  371.                 
  372.                 // Check if it is an end tag
  373.                 if (($nextTagStart >= 2) and substr_compare($theTemplateContents, self::TEMPLATE_PARSER_END_TAG, $nextTagStart-2, self::TEMPLATE_PARSER_END_TAG_LEN) == 0) {
  374.                     
  375.                     if ($inSection) {
  376.                         $section['contents'] = substr($theTemplateContents, $pos, $nextTagStart-2 - $pos);
  377.                         $sections[] = $section;
  378.                         $section = array();
  379.                         
  380.                         $inSection = false;
  381.                         $lastTagEnd = $nextTagStart-2 + self::TEMPLATE_PARSER_END_TAG_LEN;
  382.                         $pos = $lastTagEnd;
  383.                     } else {
  384.                         $section['name'] = 'END';
  385.                         $section['contents'] = substr($theTemplateContents, $pos, $nextTagStart-2 - $pos);
  386.                         $sections[] = $section;
  387.                         
  388.                         return $nextTagStart-2;
  389.                     }
  390.                     continue;
  391.                 }
  392.                 
  393.                 // Check if it is a start tag
  394.                 if (($nextTagStart >= 1) and substr_compare($theTemplateContents, self::TEMPLATE_PARSER_START_TAG, $nextTagStart-1, self::TEMPLATE_PARSER_START_TAG_LEN) == 0) {
  395.                     
  396.                     if ($inSection) {
  397.                         // Do subcall
  398.                         $section['subsections'] = array();
  399.                         $pos = $this->parseSingleSection($pos, $nextTagStart-1, $section['subsections']);
  400.                         $sections[] = $section;
  401.                         $section = array();
  402.                         
  403.                         // Skip past the end tag that had the subcall return
  404.                         $inSection = false;
  405.                         $lastTagEnd = $pos + self::TEMPLATE_PARSER_END_TAG_LEN;
  406.                         $pos = $lastTagEnd;
  407.                     } else {
  408.                         $inSection = true;
  409.                         
  410.                         $nameStart = $nextTagStart-1+self::TEMPLATE_PARSER_START_TAG_LEN;
  411.                         $nameEnd = strpos($theTemplateContents, self::TEMPLATE_PARSER_START_TAG2, $nameStart);
  412.                         if ($nameEnd === false) {
  413.                             break;    // Exception
  414.                         }
  415.  
  416.                         $name = substr($theTemplateContents, $nameStart, $nameEnd - $nameStart);
  417.                         $section['name'] = $name;
  418.                         $section['prefix'] = substr($theTemplateContents, $lastTagEnd, $nextTagStart-1 - $lastTagEnd);
  419.                         
  420.                         $pos = $nameEnd + self::TEMPLATE_PARSER_START_TAG2_LEN;
  421.                     }
  422.                     continue;
  423.                 }
  424.                 
  425.                 // not a tag
  426.                 $pos = $nextTagStart+1;
  427.             } while (true);
  428.             
  429.             return $pos;
  430.         }
